INSERTING VALUES TO POPULATE THE TABLE
TYPE1 - INSERTING THE VALUES, YOU HAVE TO ENTER THE DATA FOR EACH COLUMN
INSERT INTO Test_TBL1 VALUES(101, '101GRP1', 'GRP1', 'PSY', 92);
INSERT INTO Test_TBL1 VALUES(102, '102GRP2', 'GRP2', 'APT', 87);
INSERT INTO Test_TBL1 VALUES(103, '103GRP3', 'GRP3', 'QAPT', NULL);


TYPE2 - PROMPTING FOR INPUT
INSERT INTO Test_TBL1 VALUES(&TEST_ID, '&TEST_NAME', '&TEST_CODE', '&TEST_SUBJ_CODE', &TEST_SUBJ_MARK);

TYPE3 - INSERTING FOR SPECIFIC COLUMN- WE CAN SELECT RESPECTIVE COLUMNS IN WHICH WE WISH TO INSERT DATA
INSERT INTO Test_TBL1(TEST_ID, TEST_NAME, TEST_CODE, TEST_SUBJ_CODE, TEST_SUBJ_MARK) VALUES(102, '102GRP2', 'GRP2', 'APT', 87);

TYPE 4 - INSERT USING SELECT STATEMENT
INSERT INTO Test_TBL1 SELECT * FROM ALL_TESTS;

TYPE 4 - INSERT  SPECIFIC DATA USING SELECT STATEMENT
INSERT INTO Test_TBL1(TEST_ID, TEST_NAME) SELECT TEST_ID, TEST_NAME FROM ALL_TESTS WHERE TEST_CODE='CVL';



DML - UPDATE STATEMENT
UPDATE Test_TBL1 SET TEST_CODE='GROUP1' WHERE TEST_ID=101;

UPDATE Test_TBL1 SET TEST_SUBJ_CODE='QPT' WHERE TEST_SUBJ_MARK IS NULL;

DML - DELETE STATEMENTS
DELETE FROM Test_TBL1 WHERE TEST_SUBJ_CODE='FRN';

DELETE FROM Test_TBL1 WHERE TEST_SUBJ_MARK IS NULL;
