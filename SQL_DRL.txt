TO RETRIEVE DATA FROM DATABASE USING --> SELECT

1. RETRIEVE ALL COLUMNS
SELECT * FROM Test_TBL1;
SELECT TEST_ID, TEST_NAME, TEST_CODE, TEST_SUBJ_CODE, TEST_SUBJ_MARK FROM Test_TBL1;

2. RETRIEVAL OF PARTIAL NUMBER OF COLUMNS
SELECT TEST_ID, TEST_NAME, TEST_CODE FROM Test_TBL1;

3. TO RESTRICT THE NUMBER OF RECORDS/ROWS I.E. CONDITIONED RETRIEVAL
SELECT * FROM Test_TBL1 WHERE TEST_ID=101;

4. CONDITIONED ON NULL VALUES
SELECT TEST_NAME, TEST_SUBJ_CODE, TEST_SUBJ_MARK FROM Test_TBL1 WHERE TEST_SUBJ_MARK IS NULL;


NOTE. PARTIAL SELECTIO OF COLUMNS HAPPENS AFTER SELECT AND PARTIAL SELECTION OF ROWS HAPPEND AFTER THE WHERE CLAUSE

5. CONDITIONED ON OPERATORS[<,>,=, LIKE, BETWEEN, IN, NOT IN]
SELECT  TEST_NAME, TEST_SUBJ_CODE, TEST_SUBJ_MARK FROM Test_TBL1 WHERE TEST_SUBJ_CODE IS LIKE 'P%';
TEST_SUBJ_CODE STARTING WITH P


6. CONDITIONED AND ORDERED
SELECT  TEST_NAME, TEST_SUBJ_CODE, TEST_SUBJ_MARK FROM Test_TBL1 ORDER BY TEST_ID;

7. SUB-QUERIES - SUBQUERY EXECUTE EARLIER THAN MAIN QUERY
SELECT TEST_DOMAIN FROM ALL_TESTS WHERE TEST_CODE IN (SELECT TEST_CODE FROM Test_TBL1 WHERE TEST_ID=103);
